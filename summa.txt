<?php
// Database connection
$servername = "localhost";
$username = "root";
$password = "";
$dbname = "career_compass";

$conn = new mysqli($servername, $username, $password, $dbname);

// Check connection
if ($conn->connect_error) {
    die("Connection failed: " . $conn->connect_error);
}

// Create table query
$createTableQuery = "
CREATE TABLE IF NOT EXISTS career_combass_courses (
    id INT AUTO_INCREMENT PRIMARY KEY,
    course_id VARCHAR(10) NOT NULL,
    course VARCHAR(255) NOT NULL,
    description TEXT NOT NULL
)";
if ($conn->query($createTableQuery) === TRUE) {
    echo "Table 'career_combass_courses' created successfully.<br>";
} else {
    die("Error creating table: " . $conn->error);
}

// JSON data
$jsonData = '
[
    {
        "course_id": "bpharm",
        "course": "BPharm General Pharmacy",
        "description": "Study the fundamentals of pharmacy, drug interactions, and therapeutic management to prepare for a career in healthcare."
    },
    {
        "course_id": "bpharm",
        "course": "BPharm Pharmaceutical Chemistry",
        "description": "Learn about the chemistry of drugs, drug design, and synthesis processes for careers in pharmaceutical research and industry."
    },
    {
        "course_id": "bpharm",
        "course": "BPharm Clinical Pharmacy",
        "description": "Develop expertise in patient medication management, pharmacy ethics, and healthcare regulations for working in hospitals and clinics."
    },
    {
        "course_id": "bpharm",
        "course": "BPharm Pharmacology",
        "description": "Focus on the study of drug actions, therapeutic uses, and drug toxicity to prepare for a career in pharmacology and toxicology."
    }
]








';

// Parse JSON data
$courses = json_decode($jsonData, true);
// Check if JSON parsing was successful
if (json_last_error() !== JSON_ERROR_NONE) {
    die("Error parsing JSON: " . json_last_error_msg());
}

// Prepare insert query


foreach ($courses as $course) {
  // Prepare the query with placeholder parameters for values
  $insertQuery = $conn->prepare("
      INSERT INTO career_combass_courses (course_id, course, description)
      VALUES (?, ?, ?)
  ");

  // Check if the query was prepared successfully
  if ($insertQuery === false) {
      die("Error preparing query: " . $conn->error);
  }

  // Bind the parameters to the prepared statement
  $insertQuery->bind_param(
      "sss", // Three string parameters: course_id, course, and description
      $course['course_id'],
      $course['course'],
      $course['description']
  );

  // Execute the query
  if ($insertQuery->execute()) {
      echo "Inserted: " . $course['course'] . "<br>";
  } else {
      echo "Error inserting data: " . $insertQuery->error . "<br>";
  }
}

// Close the prepared statement and database connection
$insertQuery->close();
$conn->close();


// Close connections
$insertQuery->close();
$conn->close();
